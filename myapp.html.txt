<script>
    // تخزين البيانات في ذاكرة المتصفح المحلية
    function saveToPhone() {
        const name = document.getElementById('custName').value;
        const amount = document.getElementById('amount').value;

        if(!name || !amount) { alert("يرجى ملء الخانات"); return; }

        let sales = JSON.parse(localStorage.getItem('mySales') || "[]");
        sales.push({ name: name.trim(), amount: amount, status: "لم يحول بعد" });
        localStorage.setItem('mySales', JSON.stringify(sales));
        
        alert("تم حفظ العملية بنجاح!");
        document.getElementById('custName').value = "";
        document.getElementById('amount').value = "";
        loadTable();
    }

    function loadTable() {
        const sales = JSON.parse(localStorage.getItem('mySales') || "[]");
        const tbody = document.getElementById('tableBody');
        tbody.innerHTML = "";
        sales.forEach((item) => {
            let statusClass = item.status === "✅ تم الدفع" ? "paid" : "unpaid";
            tbody.innerHTML += `<tr><td>${item.name}</td><td>${item.amount}</td><td class="${statusClass}">${item.status}</td></tr>`;
        });
    }

    function startComparison() {
        const fileInput = document.getElementById('excelFile');
        if (!fileInput.files[0]) { alert("الرجاء اختيار ملف إكسل أولاً"); return; }

        const reader = new FileReader();
        reader.onload = function(e) {
            const data = new Uint8Array(e.target.result);
            const workbook = XLSX.read(data, {type: 'array'});
            const sheet = workbook.Sheets[workbook.SheetNames[0]];
            const excelData = XLSX.utils.sheet_to_json(sheet);

            // تحويل أسماء الإكسل لقائمة نصوص للمقارنة
            let bankNames = excelData.map(row => Object.values(row).join(" "));

            let sales = JSON.parse(localStorage.getItem('mySales') || "[]");
            
            sales.forEach(sale => {
                // البحث: هل اسم الزبون موجود داخل أي سطر في الإكسل؟
                let found = bankNames.some(bankRow => bankRow.includes(sale.name));
                if (found) {
                    sale.status = "✅ تم الدفع";
                } else {
                    sale.status = "❌ لم يحول بعد";
                }
            });

            localStorage.setItem('mySales', JSON.stringify(sales));
            loadTable();
            alert("اكتملت المقارنة!");
        };
        reader.readAsArrayBuffer(fileInput.files[0]);
    }

    // تشغيل الجدول عند فتح التطبيق
    loadTable();
</script>
